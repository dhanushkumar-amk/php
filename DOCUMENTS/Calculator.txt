<?php
$x = 10;
$y = 4;
$operator = "*";
switch ($operator) {
    case "+":
        echo "Addition " ,  $x + $y;
        break;
    case "-":
        echo "subteaction ", $x - $y;
        break;
    case "*":
        echo "Multiplication ", $x * $y;
        break;
    case "/":
        echo "Division ", $x / $y;
        break;
    case "%":
        echo "Modulues ", $x % $y;
        break;
    default:
        echo "Incorrect Operator";
        break;
}
?>
