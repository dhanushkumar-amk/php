<?php
// Define the array to be sorted
$numbers = array(3, 2, 5, 4, 1);
// Ascending order
$ascending = $numbers;
sort($ascending);
echo "Ascending order: ";
foreach($ascending as $num) {
    echo $num . " ";
}
echo "\n";
// Descending order
$descending = $numbers;
rsort($descending);
echo "Descending order: ";
foreach($descending as $num) {
    echo $num . " ";
}
echo "\n";
?>
